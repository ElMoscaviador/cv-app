const TextBox = ({ label, helpText, ...props }) => {
  const [field, meta] = useField(props);

  const [didFocus, setDidFocus] = React.useState(false);
  const handleFocus = () => setDidFocus(true);
  const showFeedback =
    (!!didFocus || meta.touched);
  const maxCharacters = 30;  

  return (
    <div
      className='textbox__wrapper'
    >
      <div className="label-feedback__wrapper">
        <label htmlFor='textbox'>{label}</label>
        {showFeedback && (
          <span
            className='feedback'
            aria-live="polite"
          >
            `${maxCharacters - field.value.length} characters left`
          </span>
        )}
      </div>
      <input
        {...props}
        {...field}
        onFocus={handleFocus}
      />
       {meta.touched && meta.error && <div className="error">{meta.error}}
    </div>
  );
};
