const InputBoxWithDropdown = ({contents}) => {
    const [selection, setSelection] = useState('')
    const [dropdown, showDropdown] = useState(false)

    const showDropdownMatch(event) {
    showDropdown(true)
    setSelection(event.target.value)
    }

    return (
       <div className='input-box-with-dropdown>
       <div className='input-box'>
       <label className='input-box__label' htmlFor={props.name}>{label}</label>
       <input className="input-box__field" 
       onBlur={props.handleBlur}
       onChange={(event) => {
          showDropdownMatch(event);
          field.onChange(event);
        }}}
       value={selection} {...props} />
       {meta.touched && meta.error && <div className="error">{meta.error}}
     </div>
     {dropdown && <div className='dropdown'>
     {contents.map((item) => {
        return item.includes(selection) ? <input className='dropdown-item' type='button' value={item} onClick={(event) => setSelection(event.target.value)} : null/>
     })}}
     </div>
     </div>
    )
}
