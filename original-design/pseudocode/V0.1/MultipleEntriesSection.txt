class EntriesManager {
render() {
    const {savedEntries} = this.props;
    return (
        <div class='entries-manager'>
        <div class='added-entries'>
        {Object.entries(savedEntries).map((entry) => {
           <CustomEntryView entry={entry}>
        })}
        </div>
        <AddExperienceButton/>
        </div>
    )
}
}

class MultipleEntriesSection {
    constructor(props) {
        this.state = {savedEntries: props.savedEntries || {}}
    }
    render() {
        const {displayNextSection, displayNextSectionAndSaveData, saveInPreview, previewMode} = this.props;
        return (
            <EntriesManager savedEntries={savedEntries}/>
            {!previewMode && !this.state.savedEntries.length && <SkipSectionButton={displayNextSection}/>}
            {!previewMode && this.state.savedEntries.length && <DisplayNextSectionButton displayNextSectionAndSaveData={displayNextSectionAndSaveData}/>}
            {previewMode && <SaveInPreviewButton saveInPreview={saveInPreview} />}
        )
    }
}
